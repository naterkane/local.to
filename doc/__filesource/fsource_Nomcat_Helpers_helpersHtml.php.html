  170.      * @param  string $insertBefore[optional] String to be inserted before options.

  174.     function _parseAttributes($options, $exclude = null, $insertBefore = ' ', $insertAfter = null) {
  175.         if (is_array($options)) {
  176.             $options = array_merge(array('escape' => true), $options);
  177.  
  178.             if (!is_array($exclude)) {
  179.                 $exclude = array();
  180.             }
  181.             $keys = array_diff(array_keys($options), array_merge((array)$exclude, array('escape')));
  182.             $values = array_intersect_key(array_values($options), $keys);
  183.             $escape = $options['escape'];
  184.             $attributes = array();
  185.  
  186.             foreach ($keys as $index => $key) {
  187.                 $attributes[] = $this->__formatAttribute($key, $values[$index], $escape);
  188.             }
  189.             $out = implode(' ', $attributes);
  190.         } else {
  191.             $out = $options;
  192.         }
  193.         return $out ? $insertBefore . $out . $insertAfter : '';
  194.     }

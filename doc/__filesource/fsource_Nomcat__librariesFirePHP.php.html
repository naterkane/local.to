  629.   
  630.     $Type = null;
  631.     $Label = null;
  632.     $Options = array();

  634.     if(func_num_args()==1) {
  635.     } else

  638.         case self::LOG:

  642.         case self::DUMP:
  643.         case self::TRACE:
  644.         case self::EXCEPTION:
  645.         case self::TABLE:
  646.         case self::GROUP_START:
  647.         case self::GROUP_END:
  648.           $Type = func_get_arg(1);
  649.           break;
  650.         default:
  651.           $Label = func_get_arg(1);
  652.           break;
  653.       }
  654.     } else
  655.     if(func_num_args()==3) {
  656.       $Type = func_get_arg(2);
  657.       $Label = func_get_arg(1);
  658.     } else
  659.     if(func_num_args()==4) {
  660.       $Type = func_get_arg(2);
  661.       $Label = func_get_arg(1);
  662.       $Options = func_get_arg(3);
  663.     } else {
  664.       throw $this->newException('Wrong number of arguments to fb() function!');
  665.     }
  666.   
  667.   
  668.     if(!$this->detectClientExtension()) {
  669.       return false;
  670.     }
  671.   
  672.     $meta = array();
  673.     $skipFinalObjectEncode = false;
  674.   
  675.     if($Object instanceof Exception) {
  676.  
  677.       $meta['file'] = $this->_escapeTraceFile($Object->getFile());
  678.       $meta['line'] = $Object->getLine();
  679.       
  680.       $trace = $Object->getTrace();
  681.       if($Object instanceof ErrorException
  682.          && isset($trace[0]['function'])
  683.          && $trace[0]['function']=='errorHandler'
  684.          && isset($trace[0]['class'])
  685.          && $trace[0]['class']=='FirePHP') {
  686.            
  687.         $severity = false;
  688.         switch($Object->getSeverity()) {
  689.           case E_WARNING: $severity = 'E_WARNING'; break;
  690.           case E_NOTICE: $severity = 'E_NOTICE'; break;
  691.           case E_USER_ERROR: $severity = 'E_USER_ERROR'; break;
  692.           case E_USER_WARNING: $severity = 'E_USER_WARNING'; break;
  693.           case E_USER_NOTICE: $severity = 'E_USER_NOTICE'; break;
  694.           case E_STRICT: $severity = 'E_STRICT'; break;
  695.           case E_RECOVERABLE_ERROR: $severity = 'E_RECOVERABLE_ERROR'; break;
  696.           case E_DEPRECATED: $severity = 'E_DEPRECATED'; break;
  697.           case E_USER_DEPRECATED: $severity = 'E_USER_DEPRECATED'; break;
  698.         }
  699.            
  700.         $Object = array('Class'=>get_class($Object),
  701.                         'Message'=>$severity.': '.$Object->getMessage(),
  702.                         'File'=>$this->_escapeTraceFile($Object->getFile()),
  703.                         'Line'=>$Object->getLine(),
  704.                         'Type'=>'trigger',
  705.                         'Trace'=>$this->_escapeTrace(array_splice($trace,2)));
  706.         $skipFinalObjectEncode = true;
  707.       } else {
  708.         $Object = array('Class'=>get_class($Object),
  709.                         'Message'=>$Object->getMessage(),
  710.                         'File'=>$this->_escapeTraceFile($Object->getFile()),
  711.                         'Line'=>$Object->getLine(),
  712.                         'Type'=>'throw',
  713.                         'Trace'=>$this->_escapeTrace($trace));
  714.         $skipFinalObjectEncode = true;
  715.       }
  716.       $Type = self::EXCEPTION;
  717.       
  718.     } else
  719.     if($Type==self::TRACE) {
  720.       
  721.       $trace = debug_backtrace();
  722.       if(!$trace) return false;
  723.       for( $i=0 ; $i<sizeof($trace) ; $i++ ) {
  724.  
  725.         if(isset($trace[$i]['class'])
  726.            && isset($trace[$i]['file'])
  727.            && ($trace[$i]['class']=='FirePHP'
  728.                || $trace[$i]['class']=='FB')
  729.            && (substr($this->_standardizePath($trace[$i]['file']),-18,18)=='FirePHPCore/fb.php'
  730.                || substr($this->_standardizePath($trace[$i]['file']),-29,29)=='FirePHPCore/FirePHP.class.php')) {
  731.           /* Skip - FB::trace(), FB::send(), $firephp->trace(), $firephp->fb() */
  732.         } else
  733.         if(isset($trace[$i]['class'])
  734.            && isset($trace[$i+1]['file'])
  735.            && $trace[$i]['class']=='FirePHP'
  736.            && substr($this->_standardizePath($trace[$i+1]['file']),-18,18)=='FirePHPCore/fb.php') {
  737.           /* Skip fb() */
  738.         } else
  739.         if($trace[$i]['function']=='fb'
  740.            || $trace[$i]['function']=='trace'
  741.            || $trace[$i]['function']=='send') {
  742.           $Object = array('Class'=>isset($trace[$i]['class'])?$trace[$i]['class']:'',
  743.                           'Type'=>isset($trace[$i]['type'])?$trace[$i]['type']:'',
  744.                           'Function'=>isset($trace[$i]['function'])?$trace[$i]['function']:'',
  745.                           'Message'=>$trace[$i]['args'][0],
  746.                           'File'=>isset($trace[$i]['file'])?$this->_escapeTraceFile($trace[$i]['file']):'',
  747.                           'Line'=>isset($trace[$i]['line'])?$trace[$i]['line']:'',
  748.                           'Args'=>isset($trace[$i]['args'])?$this->encodeObject($trace[$i]['args']):'',
  749.                           'Trace'=>$this->_escapeTrace(array_splice($trace,$i+1)));
  750.  
  751.           $skipFinalObjectEncode = true;
  752.           $meta['file'] = isset($trace[$i]['file'])?$this->_escapeTraceFile($trace[$i]['file']):'';
  753.           $meta['line'] = isset($trace[$i]['line'])?$trace[$i]['line']:'';
  754.           break;
  755.         }
  756.       }
  757.  
  758.     } else
  759.     if($Type==self::TABLE) {
  760.       
  761.       if(isset($Object[0]) && is_string($Object[0])) {
  762.         $Object[1] = $this->encodeTable($Object[1]);
  763.       } else {
  764.         $Object = $this->encodeTable($Object);
  765.       }
  766.  
  767.       $skipFinalObjectEncode = true;
  768.       
  769.     } else
  770.     if($Type==self::GROUP_START) {
  771.       
  772.       if(!$Label) {
  773.         throw $this->newException('You must specify a label for the group!');
  774.       }
  775.       
  776.     } else {
  777.       if($Type===null) {
  778.         $Type = self::LOG;
  779.       }
  780.     }
  781.     
  782.     if($this->options['includeLineNumbers']) {
  783.       if(!isset($meta['file']) || !isset($meta['line'])) {
  784.  
  785.         $trace = debug_backtrace();
  786.         for( $i=0 ; $trace && $i<sizeof($trace) ; $i++ ) {
  787.   
  788.           if(isset($trace[$i]['class'])
  789.              && isset($trace[$i]['file'])
  790.              && ($trace[$i]['class']=='FirePHP'
  791.                  || $trace[$i]['class']=='FB')
  792.              && (substr($this->_standardizePath($trace[$i]['file']),-18,18)=='FirePHPCore/fb.php'
  793.                  || substr($this->_standardizePath($trace[$i]['file']),-29,29)=='FirePHPCore/FirePHP.class.php')) {
  794.             /* Skip - FB::trace(), FB::send(), $firephp->trace(), $firephp->fb() */
  795.           } else
  796.           if(isset($trace[$i]['class'])
  797.              && isset($trace[$i+1]['file'])
  798.              && $trace[$i]['class']=='FirePHP'
  799.              && substr($this->_standardizePath($trace[$i+1]['file']),-18,18)=='FirePHPCore/fb.php') {
  800.             /* Skip fb() */
  801.           } else
  802.           if(isset($trace[$i]['file'])
  803.              && substr($this->_standardizePath($trace[$i]['file']),-18,18)=='FirePHPCore/fb.php') {
  804.             /* Skip FB::fb() */
  805.           } else {
  806.             $meta['file'] = isset($trace[$i]['file'])?$this->_escapeTraceFile($trace[$i]['file']):'';
  807.             $meta['line'] = isset($trace[$i]['line'])?$trace[$i]['line']:'';
  808.             break;
  809.           }
  810.         }      
  811.       
  812.       }
  813.     } else {
  814.       unset($meta['file']);
  815.       unset($meta['line']);
  816.     }
  817.  
  818.       $this->setHeader('X-Wf-Protocol-1','http://meta.wildfirehq.org/Protocol/JsonStream/0.2');
  819.       $this->setHeader('X-Wf-1-Plugin-1','http://meta.firephp.org/Wildfire/Plugin/FirePHP/Library-FirePHPCore/'.self::VERSION);
  820.  
  821.     $structure_index = 1;
  822.     if($Type==self::DUMP) {
  823.       $structure_index = 2;
  824.         $this->setHeader('X-Wf-1-Structure-2','http://meta.firephp.org/Wildfire/Structure/FirePHP/Dump/0.1');
  825.     } else {
  826.         $this->setHeader('X-Wf-1-Structure-1','http://meta.firephp.org/Wildfire/Structure/FirePHP/FirebugConsole/0.1');
  827.     }
  828.   
  829.     if($Type==self::DUMP) {
  830.         $msg = '{"'.$Label.'":'.$this->jsonEncode($Object, $skipFinalObjectEncode).'}';
  831.     } else {
  832.       $msg_meta = $Options;
  833.       $msg_meta['Type'] = $Type;
  834.       if($Label!==null) {
  835.         $msg_meta['Label'] = $Label;
  836.       }
  837.       if(isset($meta['file']) && !isset($msg_meta['File'])) {
  838.         $msg_meta['File'] = $meta['file'];
  839.       }
  840.       if(isset($meta['line']) && !isset($msg_meta['Line'])) {
  841.         $msg_meta['Line'] = $meta['line'];
  842.       }
  843.         $msg = '['.$this->jsonEncode($msg_meta).','.$this->jsonEncode($Object, $skipFinalObjectEncode).']';
  844.     }
  845.     
  846.     $parts = explode("\n",chunk_split($msg, 5000, "\n"));
  847.  
  848.     for( $i=0 ; $i<count($parts) ; $i++) {
  849.         
  850.         $part = $parts[$i];
  851.         if ($part) {
  852.             
  853.             if(count($parts)>2) {
  854.               // Message needs to be split into multiple parts

  857.                                . '|' . $part . '|'
  858.                                . (($i<count($parts)-2)?'\\':''));
  859.             } else {
  860.               $this->setHeader('X-Wf-1-'.$structure_index.'-'.'1-'.$this->messageIndex,
  861.                                strlen($part) . '|' . $part . '|');
  862.             }
  863.             
  864.             $this->messageIndex++;
  865.             
  866.             if ($this->messageIndex > 99999) {
  867.                 throw $this->newException('Maximum number (99,999) of messages reached!');             
  868.             }
  869.         }
  870.     }
  871.  
  872.       $this->setHeader('X-Wf-1-Index',$this->messageIndex-1);
  873.  
  874.     return true;
  875.   }

  1019.   /**
  1020.    * Encodes an object including members with
  1021.    * protected and private visibility
  1022.    * 
  1023.    * @param Object $Object The object to be encoded
  1024.    * @param int $Depth The current traversal depth
  1025.    * @return array All members of the object
  1026.    */
  1027.   protected function encodeObject($Object, $ObjectDepth = 1, $ArrayDepth = 1)
  1028.   {
  1029.     $return = array();
  1030.  
  1031.     if (is_resource($Object)) {
  1032.  
  1033.       return '** '.(string)$Object.' **';
  1034.  
  1035.     } else    
  1036.     if (is_object($Object)) {
  1037.  
  1038.         if ($ObjectDepth > $this->options['maxObjectDepth']) {
  1039.           return '** Max Object Depth ('.$this->options['maxObjectDepth'].') **';
  1040.         }
  1041.         
  1042.         foreach ($this->objectStack as $refVal) {
  1043.             if ($refVal === $Object) {
  1044.                 return '** Recursion ('.get_class($Object).') **';
  1045.             }
  1046.         }
  1047.         array_push($this->objectStack, $Object);
  1048.                 
  1049.         $return['__className'] = $class = get_class($Object);
  1050.         $class_lower = strtolower($class);
  1051.  
  1052.         $reflectionClass = new ReflectionClass($class);  
  1053.         $properties = array();
  1054.         foreach( $reflectionClass->getProperties() as $property) {
  1055.           $properties[$property->getName()] = $property;
  1056.         }
  1057.             
  1058.         $members = (array)$Object;
  1059.             
  1060.         foreach( $properties as $raw_name => $property ) {
  1061.           
  1062.           $name = $raw_name;
  1063.           if($property->isStatic()) {
  1064.             $name = 'static:'.$name;
  1065.           }
  1066.           if($property->isPublic()) {
  1067.             $name = 'public:'.$name;
  1068.           } else
  1069.           if($property->isPrivate()) {
  1070.             $name = 'private:'.$name;
  1071.             $raw_name = "\0".$class."\0".$raw_name;
  1072.           } else
  1073.           if($property->isProtected()) {
  1074.             $name = 'protected:'.$name;
  1075.             $raw_name = "\0".'*'."\0".$raw_name;
  1076.           }
  1077.           
  1078.           if(!(isset($this->objectFilters[$class_lower])
  1079.                && is_array($this->objectFilters[$class_lower])
  1080.                && in_array($raw_name,$this->objectFilters[$class_lower]))) {
  1081.  
  1082.             if(array_key_exists($raw_name,$members)
  1083.                && !$property->isStatic()) {
  1084.               
  1085.               $return[$name] = $this->encodeObject($members[$raw_name], $ObjectDepth + 1, 1);      
  1086.             
  1087.             } else {
  1088.               if(method_exists($property,'setAccessible')) {
  1089.                 $property->setAccessible(true);
  1090.                 $return[$name] = $this->encodeObject($property->getValue($Object), $ObjectDepth + 1, 1);
  1091.               } else
  1092.               if($property->isPublic()) {
  1093.                 $return[$name] = $this->encodeObject($property->getValue($Object), $ObjectDepth + 1, 1);
  1094.               } else {
  1095.                 $return[$name] = '** Need PHP 5.3 to get value **';
  1096.               }
  1097.             }
  1098.           } else {
  1099.             $return[$name] = '** Excluded by Filter **';
  1100.           }
  1101.         }
  1102.         
  1103.         // Include all members that are not defined in the class
  1104.         // but exist in the object
  1105.         foreach( $members as $raw_name => $value ) {
  1106.           
  1107.           $name = $raw_name;
  1108.

  1110.             $parts = explode("\0", $name);
  1111.             $name = $parts[2];
  1112.           }
  1113.

  1115.             $name = 'undeclared:'.$name;
  1116.               
  1117.             if(!(isset($this->objectFilters[$class_lower])
  1118.                  && is_array($this->objectFilters[$class_lower])
  1119.                  && in_array($raw_name,$this->objectFilters[$class_lower]))) {
  1120.               
  1121.               $return[$name] = $this->encodeObject($value, $ObjectDepth + 1, 1);
  1122.             } else {
  1123.               $return[$name] = '** Excluded by Filter **';
  1124.             }
  1125.           }
  1126.         }
  1127.         
  1128.         array_pop($this->objectStack);
  1129.         
  1130.     } elseif (is_array($Object)) {
  1131.  
  1132.         if ($ArrayDepth > $this->options['maxArrayDepth']) {
  1133.           return '** Max Array Depth ('.$this->options['maxArrayDepth'].') **';
  1134.         }
  1135.       
  1136.         foreach ($Object as $key => $val) {
  1137.

  1139.           // if the recursion is not reset here as it contains
  1140.           // a reference to itself. This is the only way I have come up

  1142.           if($key=='GLOBALS'

  1144.              && array_key_exists('GLOBALS',$val)) {
  1145.             $val['GLOBALS'] = '** Recursion (GLOBALS) **';
  1146.           }
  1147.           
  1148.           $return[$key] = $this->encodeObject($val, 1, $ArrayDepth + 1);
  1149.         }
  1150.     } else {
  1151.       if(self::is_utf8($Object)) {
  1152.         return $Object;
  1153.       } else {
  1154.         return utf8_encode($Object);
  1155.       }
  1156.     }
  1157.     return $return;
  1158.   }

  1160.   /**
  1161.    * Returns true if $string is valid UTF-8 and false otherwise.
  1162.    *
  1163.    * @param mixed $str String to be tested
  1164.    * @return boolean
  1165.    */
  1166.   protected static function is_utf8($str) {
  1167.     $c=0; $b=0;
  1168.     $bits=0;
  1169.     $len=strlen($str);
  1170.     for($i=0; $i<$len; $i++){
  1171.         $c=ord($str[$i]);
  1172.         if($c > 128){
  1173.             if(($c >= 254)) return false;
  1174.             elseif($c >= 252) $bits=6;
  1175.             elseif($c >= 248) $bits=5;
  1176.             elseif($c >= 240) $bits=4;
  1177.             elseif($c >= 224) $bits=3;
  1178.             elseif($c >= 192) $bits=2;
  1179.             else return false;
  1180.             if(($i+$bits) > $len) return false;
  1181.             while($bits > 1){
  1182.                 $i++;
  1183.                 $b=ord($str[$i]);
  1184.                 if($b < 128 || $b > 191) return false;
  1185.                 $bits--;
  1186.             }
  1187.         }
  1188.     }
  1189.     return true;
  1190.   }

  1299.  /**
  1300.   * encodes an arbitrary variable into JSON format
  1301.   *
  1302.   * @param    mixed   $var    any number, boolean, string, array, or object to be encoded.
  1303.   *                           see argument 1 to Services_JSON() above for array-parsing behavior.
  1304.   *                           if var is a strng, note that encode() always expects it
  1305.   *                           to be in ASCII or UTF-8 format!
  1306.   *
  1307.   * @return   mixed   JSON string representation of input var or an error if a problem occurs
  1308.   * @access   public
  1309.   */
  1310.   private function json_encode($var)
  1311.   {
  1312.     
  1313.     if(is_object($var)) {
  1314.       if(in_array($var,$this->json_objectStack)) {
  1315.         return '"** Recursion **"';
  1316.       }
  1317.     }
  1318.           
  1319.       switch (gettype($var)) {
  1320.           case 'boolean':
  1321.               return $var ? 'true' : 'false';
  1322.  
  1323.           case 'NULL':
  1324.               return 'null';
  1325.  
  1326.           case 'integer':
  1327.               return (int) $var;
  1328.  
  1329.           case 'double':
  1330.           case 'float':
  1331.               return (float) $var;
  1332.  
  1333.           case 'string':
  1334.               // STRINGS ARE EXPECTED TO BE IN ASCII OR UTF-8 FORMAT
  1335.               $ascii = '';
  1336.               $strlen_var = strlen($var);
  1337.  
  1338.              /*
  1339.               * Iterate over every character in the string,
  1340.               * escaping with a slash or encoding to UTF-8 where necessary
  1341.               */
  1342.               for ($c = 0; $c < $strlen_var; ++$c) {
  1343.  
  1344.                   $ord_var_c = ord($var{$c});
  1345.  
  1346.                   switch (true) {
  1347.                       case $ord_var_c == 0x08:
  1348.                           $ascii .= '\b';
  1349.                           break;
  1350.                       case $ord_var_c == 0x09:
  1351.                           $ascii .= '\t';
  1352.                           break;
  1353.                       case $ord_var_c == 0x0A:
  1354.                           $ascii .= '\n';
  1355.                           break;
  1356.                       case $ord_var_c == 0x0C:
  1357.                           $ascii .= '\f';
  1358.                           break;
  1359.                       case $ord_var_c == 0x0D:
  1360.                           $ascii .= '\r';
  1361.                           break;
  1362.  
  1363.                       case $ord_var_c == 0x22:
  1364.                       case $ord_var_c == 0x2F:
  1365.                       case $ord_var_c == 0x5C:
  1366.                           // double quote, slash, slosh
  1367.                           $ascii .= '\\'.$var{$c};
  1368.                           break;
  1369.  
  1370.                       case (($ord_var_c >= 0x20) && ($ord_var_c <= 0x7F)):
  1371.                           // characters U-00000000 - U-0000007F (same as ASCII)
  1372.                           $ascii .= $var{$c};
  1373.                           break;
  1374.  
  1375.                       case (($ord_var_c & 0xE0) == 0xC0):
  1376.                           // characters U-00000080 - U-000007FF, mask 110XXXXX
  1377.                           // see http://www.cl.cam.ac.uk/~mgk25/unicode.html#utf-8

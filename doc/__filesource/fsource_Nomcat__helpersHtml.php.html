  152.      *    + 'compact'

  155.      *    + 'readonly'

  159.      *    + 'ismap'
  160.      *    + 'nohref'
  161.      *    + 'noshade'

  163.      *    + 'multiple'

  166.      * And its value is one of:
  167.      *    + 1

  169.      *    + 'true'

  171.      * Then the value will be reset to be identical with key's name.

  173.      *
  174.      * @param  array  $options Array of options.
  175.      * @param  array  $exclude[optional] Array of options to be excluded.
  176.      * @param  string $insertBefore[optional] String to be inserted before options.
  177.      * @param  string $insertAfter[optional]  String to be inserted ater options.

  180.     function _parseAttributes($options, $exclude = null, $insertBefore = ' ', $insertAfter = null) {
  181.         if (is_array($options)) {
  182.             $options = array_merge(array('escape' => true), $options);
  183.  
  184.             if (!is_array($exclude)) {
  185.                 $exclude = array();
  186.             }
  187.             $keys = array_diff(array_keys($options), array_merge((array)$exclude, array('escape')));
  188.             $values = array_intersect_key(array_values($options), $keys);
  189.             $escape = $options['escape'];
  190.             $attributes = array();
  191.  
  192.             foreach ($keys as $index => $key) {
  193.                 $attributes[] = $this->__formatAttribute($key, $values[$index], $escape);
  194.             }
  195.             $out = implode(' ', $attributes);
  196.         } else {
  197.             $out = $options;
  198.         }
  199.         return $out ? $insertBefore . $out . $insertAfter : '';
  200.     }
